def doc(type,regexp,&block)
  s = regexp.inspect.gsub(%r^\/\^?/,'').gsub(%r\$?\/$/,'').split(%r\([^)]*\)/)
  params = []
  unless block.nil?
    params = block.parameters.map {|_| ':' + _[1].to_s + ':' }
  end
  string = ''
  if s.size > 1
    s.each do |item|
      string << item.color(:white)
      param = params.shift
      string << param.color(:blue) if param
    end
  else
    string = s.first.color(:white)
  end
  params = params.join(',')
  params = "\n    #{params}".color(:cyan) unless params.strip.length == 0
  if string =~ %r:$/
    string << ' '
  end
  printf("%5s %s%s\n",type.to_s.color(:red),string,params)
end

def method_missing(sym,*args,&block)
  if sym.to_s =~ %r^[A-Z]/
    doc(sym,args.first,&block)
  else
    super
  end
end
